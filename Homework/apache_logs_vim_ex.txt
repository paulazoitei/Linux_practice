Install Apache and deploy "hello world" page


paulazoitei@ubuntu2404:~$ sudo apt update && sudo apt upgrade

paulazoitei@ubuntu2404:~$ sudo apt install apache2 -y

paulazoitei@ubuntu2404:~$ sudo systemctl enable apache2

paulazoitei@ubuntu2404:~$ sudo mkdir /var/www/html/test_website/

paulazoitei@ubuntu2404:~$ echo '<html><head><title>Example</title></head><body><h1>Test</h1><p>Hello World!</p></body></html>' | sudo tee /var/www/html/test_website/index.html
<html><head><title>Example</title></head><body><h1>Test</h1><p>Hello World!</p></body></html>

paulazoitei@ubuntu2404:~$ sudo tee /etc/apache2/sites-available/web.conf > /dev/null <<EOF
<VirtualHost *:80>
    ServerName web.testingserver.com
    DocumentRoot /var/www/html/website
    DirectoryIndex index.html
    ErrorLog \${APACHE_LOG_DIR}/example.com_error.log
    CustomLog \${APACHE_LOG_DIR}/example.com_requests.log combined
</VirtualHost>
EOF

paulazoitei@ubuntu2404:~$ sudo a2ensite web.conf

paulazoitei@ubuntu2404:~$ sudo systemctl reload apache2

paulazoitei@ubuntu2404:~$ sudo chown -R www-data:www-data /var/www/html/test_website

paulazoitei@ubuntu2404:~$ sudo chmod -R 755 /var/www/html/test_website
